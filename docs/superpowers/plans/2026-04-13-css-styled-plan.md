# css-styled Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a three-crate Rust workspace that provides typed, spec-validated CSS generation from theme structs for component libraries, with optional Leptos integration.

**Architecture:** Three crates — `css-spec-data` (build-script-generated CSS spec lookup tables from `@webref/css`), `css-styled-derive` (proc macro for `#[derive(StyledComponent)]` and `css!` macro), and `css-styled` (public API with `IntoCss` trait, runtime validation, and optional Leptos `StyleTag` component). The spec data crate powers compile-time validation in the derive and runtime validation in the runtime crate.

**Tech Stack:** Rust, proc-macro2/syn/quote (derive macro), serde_json (spec data parsing), ureq (HTTP fetch in build script), strsim (fuzzy matching for error suggestions), trybuild (compile-fail tests), leptos 0.8 (optional feature)

- [ ] **Step 3: Write the build script**

The build script downloads the `@webref/css` JSON index, then fetches each spec file. It parses CSS property definitions and pseudo-class/element lists, then generates a Rust source file.

```rust
// css-spec-data/build.rs
use serde::Deserialize;
use std::collections::{BTreeMap, BTreeSet};
use std::fs;
use std::io::Write;
use std::path::{Path, PathBuf};

#[derive(Deserialize)]
struct SpecFile {
    #[serde(default)]
    properties: Vec<PropertyEntry>,
    #[serde(default)]
    selectors: Vec<SelectorEntry>,
}

#[derive(Deserialize)]
struct PropertyEntry {
    name: String,
    #[serde(default)]
    value: String,
    #[serde(default, rename = "initial")]
    initial: String,
    #[serde(default)]
    inherited: InheritedValue,
}

#[derive(Deserialize, Default)]
#[serde(untagged)]
enum InheritedValue {
    #[default]
    Unknown,
    Bool(bool),
    Str(String),
}

impl InheritedValue {
    fn as_bool(&self) -> bool {
        match self {
            InheritedValue::Bool(b) => *b,
            InheritedValue::Str(s) => s == "yes",
            InheritedValue::Unknown => false,
        }
    }
}

#[derive(Deserialize)]
struct SelectorEntry {
    name: String,
}

const WEBREF_BASE: &str = "https://raw.githubusercontent.com/w3c/webref/curated/ed/css";

// Known CSS spec file names that contain property definitions
const SPEC_FILES: &[&str] = &[
    "CSS", "css-align", "css-animations", "css-backgrounds", "css-borders",
    "css-box", "css-break", "css-cascade", "css-color", "css-contain",
    "css-content", "css-display", "css-flexbox", "css-fonts", "css-grid",
    "css-images", "css-inline", "css-lists", "css-logical", "css-masking",
    "css-multicol", "css-overflow", "css-overscroll", "css-page",
    "css-position", "css-shapes", "css-sizing", "css-speech", "css-tables",
    "css-text", "css-text-decor", "css-transforms", "css-transitions",
    "css-ui", "css-values", "css-variables", "css-view-transitions",
    "css-will-change", "css-writing-modes", "filter-effects",
    "compositing", "motion", "scroll-animations",
];

// Known selectors spec
const SELECTORS_SPEC: &str = "selectors";

fn main() {
    let out_dir = std::env::var("OUT_DIR").unwrap();
    let data_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("data");
    fs::create_dir_all(&data_dir).ok();

    let mut properties: BTreeMap<String, (String, String, bool)> = BTreeMap::new();
    let mut pseudo_classes: BTreeSet<String> = BTreeSet::new();
    let mut pseudo_elements: BTreeSet<String> = BTreeSet::new();

    // Fetch and parse each spec file
    for spec_name in SPEC_FILES {
        let json = fetch_or_cache(&data_dir, spec_name);
        if let Ok(spec) = serde_json::from_str::<SpecFile>(&json) {
            for prop in spec.properties {
                if !prop.name.starts_with("--") && !prop.name.starts_with("<") {
                    properties.entry(prop.name.clone()).or_insert_with(|| {
                        (prop.value.clone(), prop.initial.clone(), prop.inherited.as_bool())
                    });
                }
            }
        }
    }

    // Fetch selectors
    let selectors_json = fetch_or_cache(&data_dir, SELECTORS_SPEC);
    if let Ok(spec) = serde_json::from_str::<SpecFile>(&selectors_json) {
        for sel in spec.selectors {
            if sel.name.starts_with("::") {
                pseudo_elements.insert(sel.name);
            } else if sel.name.starts_with(":") {
                pseudo_classes.insert(sel.name);
            }
        }
    }

    // Add common pseudo-classes/elements that might be missing
    for pc in &[":hover", ":focus", ":active", ":visited", ":link",
                ":first-child", ":last-child", ":disabled", ":enabled",
                ":checked", ":empty", ":root", ":not()", ":is()", ":has()",
                ":where()", ":nth-child()", ":nth-last-child()",
                ":first-of-type", ":last-of-type", ":focus-visible",
                ":focus-within", ":placeholder-shown"] {
        pseudo_classes.insert(pc.to_string());
    }
    for pe in &["::before", "::after", "::first-line", "::first-letter",
                "::placeholder", "::selection", "::backdrop", "::marker",
                "::file-selector-button"] {
        pseudo_elements.insert(pe.to_string());
    }

    // Generate Rust source
    let dest = Path::new(&out_dir).join("generated.rs");
    let mut f = fs::File::create(&dest).unwrap();

    // Property table
    writeln!(f, "use crate::PropertySpec;").unwrap();
    writeln!(f, "pub static PROPERTIES: &[PropertySpec] = &[").unwrap();
    for (name, (syntax, initial, inherited)) in &properties {
        let syntax_escaped = syntax.replace('\\', "\\\\").replace('"', "\\\"");
        let initial_escaped = initial.replace('\\', "\\\\").replace('"', "\\\"");
        writeln!(f, "    PropertySpec {{ name: \"{name}\", syntax: \"{syntax_escaped}\", initial: \"{initial_escaped}\", inherited: {inherited} }},").unwrap();
    }
    writeln!(f, "];").unwrap();

    // Property names array
    writeln!(f, "pub static PROPERTY_NAMES: &[&str] = &[").unwrap();
    for name in properties.keys() {
        writeln!(f, "    \"{name}\",").unwrap();
    }
    writeln!(f, "];").unwrap();

    // Pseudo-classes array (strip parenthesized parts for matching)
    writeln!(f, "pub static PSEUDO_CLASSES: &[&str] = &[").unwrap();
    for name in &pseudo_classes {
        let base = name.split('(').next().unwrap_or(name);
        writeln!(f, "    \"{base}\",").unwrap();
    }
    writeln!(f, "];").unwrap();

    // Pseudo-elements array
    writeln!(f, "pub static PSEUDO_ELEMENTS: &[&str] = &[").unwrap();
    for name in &pseudo_elements {
        let base = name.split('(').next().unwrap_or(name);
        writeln!(f, "    \"{base}\",").unwrap();
    }
    writeln!(f, "];").unwrap();

    println!("cargo::rerun-if-changed=build.rs");
    println!("cargo::rerun-if-changed=data/");
}

fn fetch_or_cache(data_dir: &Path, spec_name: &str) -> String {
    let cache_file = data_dir.join(format!("{spec_name}.json"));

    // Try cache first
    if let Ok(cached) = fs::read_to_string(&cache_file) {
        return cached;
    }

    // Fetch from web
    let url = format!("{WEBREF_BASE}/{spec_name}.json");
    match ureq::get(&url).call() {
        Ok(resp) => {
            let body = resp.into_body().read_to_string().unwrap_or_default();
            fs::write(&cache_file, &body).ok();
            body
        }
        Err(e) => {
            eprintln!("cargo:warning=Failed to fetch {url}: {e}");
            "{}".to_string()
        }
    }
}
```

```rust
// css-spec-data/src/validate.rs

use crate::{property, ValidationResult};

/// Validate a CSS value for a given property.
pub fn validate_value(prop_name: &str, value: &str) -> ValidationResult {
    let value = value.trim();

    // Global keywords are always valid
    if matches!(value, "inherit" | "initial" | "unset" | "revert" | "revert-layer") {
        return ValidationResult::Valid;
    }

    // CSS functions pass through — can't statically validate
    if value.starts_with("var(")
        || value.starts_with("calc(")
        || value.starts_with("env(")
        || value.starts_with("min(")
        || value.starts_with("max(")
        || value.starts_with("clamp(")
    {
        return ValidationResult::Valid;
    }

    // Look up the property
    let Some(spec) = property(prop_name) else {
        // Unknown property — can't validate, pass through
        return ValidationResult::Valid;
    };

    let syntax = spec.syntax;

    // Classify what the value looks like
    let looks_like = classify_value(value);

    // Check if the value type is plausible for this property's syntax
    match looks_like {
        ValueType::Length => {
            if syntax_accepts_length(syntax) {
                ValidationResult::Valid
            } else {
                ValidationResult::Warn(format!(
                    "property '{}' has syntax '{}' — '{}' looks like a length, which may not be valid here",
                    prop_name, syntax, value
                ))
            }
        }
        ValueType::Percentage => {
            if syntax_accepts_percentage(syntax) || syntax_accepts_length(syntax) {
                ValidationResult::Valid
            } else {
                ValidationResult::Warn(format!(
                    "property '{}' — '{}' looks like a percentage, which may not be valid here",
                    prop_name, value
                ))
            }
        }
        ValueType::Color => {
            if syntax_accepts_color(syntax) {
                ValidationResult::Valid
            } else {
                ValidationResult::Warn(format!(
                    "property '{}' has syntax '{}' — '{}' looks like a color, which may not be valid here",
                    prop_name, syntax, value
                ))
            }
        }
        ValueType::Number => {
            // Numbers are valid in many contexts (flex, opacity, z-index, etc.)
            ValidationResult::Valid
        }
        ValueType::Keyword => {
            // Keywords are hard to fully validate without parsing the full syntax grammar.
            // Accept them — CSS is permissive with keywords.
            ValidationResult::Valid
        }
        ValueType::Function => {
            // Non-var/calc functions (rgb, hsl, etc.) — accept
            ValidationResult::Valid
        }
        ValueType::Compound => {
            // Multi-value (shorthand-like): "1px solid #333", "background 0.1s ease"
            ValidationResult::Valid
        }
    }
}

#[derive(Debug)]
enum ValueType {
    Length,      // 10px, 2em, 100vh, 0
    Percentage,  // 50%
    Color,       // #fff, #007acc, rgb(), hsl(), named colors, transparent, currentColor
    Number,      // 0, 1, 0.5
    Keyword,     // auto, none, flex, inherit, etc.
    Function,    // rgb(), hsl(), etc.
    Compound,    // multi-part values (shorthands)
}

fn classify_value(value: &str) -> ValueType {
    // Multi-part value (contains spaces not inside parens)
    if has_top_level_space(value) {
        return ValueType::Compound;
    }

    // Function call
    if value.contains('(') && value.ends_with(')') {
        if is_color_function(value) {
            return ValueType::Color;
        }
        return ValueType::Function;
    }

    // Hex color
    if value.starts_with('#') && value.len() <= 9 && value[1..].chars().all(|c| c.is_ascii_hexdigit()) {
        return ValueType::Color;
    }

    // Named color or color keyword
    if is_named_color(value) {
        return ValueType::Color;
    }

    // Percentage
    if value.ends_with('%') && value[..value.len()-1].parse::<f64>().is_ok() {
        return ValueType::Percentage;
    }

    // Length with unit
    let length_units = ["px", "em", "rem", "vh", "vw", "vmin", "vmax", "ch", "ex",
                        "cm", "mm", "in", "pt", "pc", "svh", "svw", "dvh", "dvw",
                        "lvh", "lvw", "cqw", "cqh"];
    for unit in &length_units {
        if value.ends_with(unit) {
            let num_part = &value[..value.len() - unit.len()];
            if num_part.parse::<f64>().is_ok() {
                return ValueType::Length;
            }
        }
    }

    // Plain number (could be 0 as a length, or a unitless number like opacity)
    if value.parse::<f64>().is_ok() {
        if value == "0" {
            return ValueType::Length; // 0 is valid as a length without units
        }
        return ValueType::Number;
    }

    // Everything else is a keyword
    ValueType::Keyword
}

fn has_top_level_space(value: &str) -> bool {
    let mut depth = 0;
    for c in value.chars() {
        match c {
            '(' => depth += 1,
            ')' => depth -= 1,
            ' ' if depth == 0 => return true,
            _ => {}
        }
    }
    false
}
```

```rust
/// Parsed struct-level component attributes.
#[derive(Debug)]
pub struct ComponentAttrs {
    pub scope: String,
    pub scope_span: proc_macro2::Span,
    pub classes: Vec<(String, String)>,  // (alias, css_name)
    pub modifiers: Vec<String>,
}

/// Parsed field-level prop attributes.
#[derive(Debug)]
pub struct PropAttr {
    pub css_property: String,
    pub css_property_span: proc_macro2::Span,
    pub on: Option<String>,
    pub on_span: Option<proc_macro2::Span>,
    pub pseudo: Option<String>,
    pub pseudo_span: Option<proc_macro2::Span>,
}

/// A parsed field: either a prop mapping or skipped.
#[derive(Debug)]
pub enum FieldKind {
    Prop(PropAttr),
    Skip,
}

/// Parse all #[component(...)] attributes from the struct.
pub fn parse_component_attrs(attrs: &[Attribute]) -> Result<ComponentAttrs> {
    let mut scope: Option<(String, proc_macro2::Span)> = None;
    let mut classes: Vec<(String, String)> = Vec::new();
    let mut modifiers: Vec<String> = Vec::new();

    for attr in attrs {
        if !attr.path().is_ident("component") {
            continue;
        }

        attr.parse_nested_meta(|meta| {
            if meta.path.is_ident("scope") {
                let _eq: Token![=] = meta.input.parse()?;
                let lit: LitStr = meta.input.parse()?;
                scope = Some((lit.value(), lit.span()));
                Ok(())
            } else if meta.path.is_ident("class") {
                let content;
                syn::parenthesized!(content in meta.input);
                while !content.is_empty() {
                    let alias: Ident = content.parse()?;
                    let _eq: Token![=] = content.parse()?;
                    let name: LitStr = content.parse()?;
                    classes.push((alias.to_string(), name.value()));
                    if content.peek(Token![,]) {
                        let _: Token![,] = content.parse()?;
                    }
                }
                Ok(())
            } else if meta.path.is_ident("modifier") {
                let content;
                syn::parenthesized!(content in meta.input);
                while !content.is_empty() {
                    let name: Ident = content.parse()?;
                    modifiers.push(name.to_string());
                    if content.peek(Token![,]) {
                        let _: Token![,] = content.parse()?;
                    }
                }
                Ok(())
            } else {
                Err(meta.error("unknown component attribute"))
            }
        })?;
    }

    let (scope_str, scope_span) = scope.ok_or_else(|| {
        Error::new(proc_macro2::Span::call_site(), "missing #[component(scope = \"...\")]")
    })?;

    Ok(ComponentAttrs {
        scope: scope_str,
        scope_span,
        classes,
        modifiers,
    })
}

/// Parse a #[prop(...)] attribute from a field.
pub fn parse_prop_attr(attrs: &[Attribute]) -> Result<Option<FieldKind>> {
    let mut found = false;

    for attr in attrs {
        if !attr.path().is_ident("prop") {
            continue;
        }
        found = true;

        let mut css_property: Option<(String, proc_macro2::Span)> = None;
        let mut on: Option<(String, proc_macro2::Span)> = None;
        let mut pseudo: Option<(String, proc_macro2::Span)> = None;
        let mut skip = false;

        attr.parse_nested_meta(|meta| {
            if meta.path.is_ident("css") {
                let _eq: Token![=] = meta.input.parse()?;
                let lit: LitStr = meta.input.parse()?;
                css_property = Some((lit.value(), lit.span()));
                Ok(())
            } else if meta.path.is_ident("on") {
                let _eq: Token![=] = meta.input.parse()?;
                let ident: Ident = meta.input.parse()?;
                on = Some((ident.to_string(), ident.span()));
                Ok(())
            } else if meta.path.is_ident("pseudo") {
                let _eq: Token![=] = meta.input.parse()?;
                let lit: LitStr = meta.input.parse()?;
                pseudo = Some((lit.value(), lit.span()));
                Ok(())
            } else if meta.path.is_ident("skip") {
                skip = true;
                Ok(())
            } else {
                Err(meta.error("unknown prop attribute"))
            }
        })?;

        if skip {
            return Ok(Some(FieldKind::Skip));
        }

        let (css_prop, css_span) = css_property.ok_or_else(|| {
            Error::new_spanned(attr, "missing css = \"...\" in #[prop(...)]")
        })?;

        return Ok(Some(FieldKind::Prop(PropAttr {
            css_property: css_prop,
            css_property_span: css_span,
            on: on.as_ref().map(|(s, _)| s.clone()),
            on_span: on.map(|(_, s)| s),
            pseudo: pseudo.as_ref().map(|(s, _)| s.clone()),
            pseudo_span: pseudo.map(|(_, s)| s),
        })));
    }

    if found {
        unreachable!()
    }
    Ok(None)
}
```

```rust
/// Find the closest match to `input` from `candidates` using Levenshtein distance.
/// Returns None if no candidate is within a reasonable distance.
pub fn closest_match<'a>(input: &str, candidates: &[&'a str]) -> Option<&'a str> {
    let mut best: Option<(&str, usize)> = None;
    let max_distance = (input.len() / 3).max(2);

    for &candidate in candidates {
        let dist = strsim::levenshtein(input, candidate);
        if dist <= max_distance {
            if best.is_none() || dist < best.unwrap().1 {
                best = Some((candidate, dist));
            }
        }
    }

    best.map(|(s, _)| s)
}
```

```rust
use quote::{quote, format_ident};
use syn::{DeriveInput, Error, Result};

use crate::parse_attrs::{parse_component_attrs, parse_prop_attr, FieldKind, PropAttr};
use crate::fuzzy::closest_match;

pub fn derive(input: DeriveInput) -> Result<TokenStream> {
    let struct_name = &input.ident;
    let comp = parse_component_attrs(&input.attrs)?;
    let scope = &comp.scope;

    // Build known alias names for validation
    let known_aliases: Vec<(&str, &str)> = comp.classes.iter()
        .map(|(alias, css)| (alias.as_str(), css.as_str()))
        .collect();

    // Parse fields
    let fields = match &input.data {
        syn::Data::Struct(data) => match &data.fields {
            syn::Fields::Named(f) => &f.named,
            _ => return Err(Error::new_spanned(&input, "StyledComponent requires named fields")),
        },
        _ => return Err(Error::new_spanned(&input, "StyledComponent can only be derived for structs")),
    };

    let mut prop_fields: Vec<(syn::Ident, PropAttr)> = Vec::new();

    for field in fields {
        let field_name = field.ident.as_ref().unwrap();
        match parse_prop_attr(&field.attrs)? {
            Some(FieldKind::Prop(prop)) => {
                // Validate CSS property name
                if css_spec_data::property(&prop.css_property).is_none() {
                    let all_names = css_spec_data::all_property_names();
                    let suggestion = closest_match(&prop.css_property, all_names);
                    let msg = if let Some(s) = suggestion {
                        format!("unknown CSS property \"{}\"\n\n  = help: did you mean \"{}\"?", prop.css_property, s)
                    } else {
                        format!("unknown CSS property \"{}\"", prop.css_property)
                    };
                    return Err(Error::new(prop.css_property_span, msg));
                }

                // Validate pseudo if present
                if let Some(ref pseudo) = prop.pseudo {
                    let is_valid = css_spec_data::is_pseudo_class(pseudo)
                        || css_spec_data::is_pseudo_element(pseudo);
                    if !is_valid {
                        let all_pseudos: Vec<&str> = css_spec_data::all_pseudo_classes().iter()
                            .chain(css_spec_data::all_pseudo_elements().iter())
                            .copied()
                            .collect();
                        let suggestion = closest_match(pseudo, &all_pseudos);
                        let msg = if let Some(s) = suggestion {
                            format!("unknown CSS pseudo-class/element \"{}\"\n\n  = help: did you mean \"{}\"?", pseudo, s)
                        } else {
                            format!("unknown CSS pseudo-class/element \"{}\"", pseudo)
                        };
                        return Err(Error::new(prop.pseudo_span.unwrap(), msg));
                    }
                }

                // Validate `on` references a known alias
                if let Some(ref on_alias) = prop.on {
                    if !known_aliases.iter().any(|(alias, _)| alias == on_alias) {
                        let available: Vec<&str> = known_aliases.iter().map(|(a, _)| *a).collect();
                        let msg = format!(
                            "unknown class alias \"{}\"\n\n  = help: available names: {}",
                            on_alias,
                            available.join(", ")
                        );
                        return Err(Error::new(prop.on_span.unwrap(), msg));
                    }
                }

                prop_fields.push((field_name.clone(), prop));
            }
            Some(FieldKind::Skip) => {}
            None => {
                return Err(Error::new_spanned(
                    field,
                    format!(
                        "field `{}` has no #[prop(...)] attribute\n\n  = help: add #[prop(css = \"...\")] or #[prop(skip)] to this field",
                        field_name
                    ),
                ));
            }
        }
    }

    // Check for duplicate (css_property, on, pseudo) combinations
    let mut seen: Vec<(&str, &Option<String>, &Option<String>)> = Vec::new();
    for (field_name, prop) in &prop_fields {
        let key = (prop.css_property.as_str(), &prop.on, &prop.pseudo);
        if seen.iter().any(|s| s.0 == key.0 && s.1 == key.1 && s.2 == key.2) {
            return Err(Error::new(
                prop.css_property_span,
                format!(
                    "duplicate CSS rule: property \"{}\" on {:?} pseudo {:?} is already defined by another field",
                    prop.css_property, prop.on, prop.pseudo
                ),
            ));
        }
        seen.push(key);
    }

    // --- Code generation ---

    // 1. Constants for scope and classes
    let scope_lit = &comp.scope;
    let class_consts: Vec<TokenStream> = comp.classes.iter().map(|(alias, css_name)| {
        let const_name = format_ident!("{}", alias.to_uppercase());
        quote! { pub const #const_name: &'static str = #css_name; }
    }).collect();

    // 2. Modifier enum + class() method
    let modifier_enum_name = format_ident!("{}Modifier", struct_name);
    let modifier_variants: Vec<proc_macro2::Ident> = comp.modifiers.iter().map(|m| {
        format_ident!("{}", to_pascal_case(m))
    }).collect();
    let modifier_strings: Vec<&String> = comp.modifiers.iter().collect();

    let modifier_enum = if comp.modifiers.is_empty() {
        quote! {}
    } else {
        quote! {
            #[derive(Debug, Clone, Copy, PartialEq, Eq)]
            pub enum #modifier_enum_name {
                #(#modifier_variants),*
            }

            impl #modifier_enum_name {
                pub fn as_str(&self) -> &'static str {
                    match self {
                        #(#modifier_enum_name::#modifier_variants => #modifier_strings),*
                    }
                }
            }
        }
    };

    let class_method = if comp.modifiers.is_empty() {
        quote! {
            pub fn class(_modifiers: &[()]) -> String {
                #scope_lit.to_string()
            }
        }
    } else {
        quote! {
            pub fn class(modifiers: &[#modifier_enum_name]) -> String {
                let mut result = String::from(#scope_lit);
                for m in modifiers {
                    result.push(' ');
                    result.push_str(m.as_str());
                }
                result
            }
        }
    };

    // Build CSS rule generation code
    let css_rule_parts: Vec<TokenStream> = prop_fields.iter().map(|(field_name, prop)| {

                        web_sys::console::warn_1(&format!("css-styled invalid: {}", msg).into());
                        #[cfg(not(target_arch = "wasm32"))]
                        eprintln!("css-styled invalid: {}", msg);
                    }
                    _ => {}
                }
            }
            rules.push(format!("{} {{ {}: {}; }}", #selector, #css_prop, &self.#field_name));
        }
    }).collect();

    let into_css_impl = quote! {
        impl css_styled::IntoCss for #struct_name {
            fn to_css(&self) -> String {
                let base = <Self as css_styled::StyledComponentBase>::base_css();
                let mut rules: Vec<String> = Vec::new();

                #(#css_rule_parts)*

                if base.is_empty() {
                    rules.join("\n")
                } else {
                    format!("{}\n{}", base, rules.join("\n"))
                }
            }

            fn scope(&self) -> &'static str {
                #scope_lit
            }
        }

        impl css_styled::StyledComponentBase for #struct_name {}
    };

    let expanded = quote! {
        impl #struct_name {
            pub const SCOPE: &'static str = #scope_lit;
            #(#class_consts)*

            #class_method
        }

        #modifier_enum

        #into_css_impl
    };

    Ok(expanded)
}

fn build_selector(
    scope: &str,
    on: &Option<String>,
    pseudo: &Option<String>,
    classes: &[(String, String)],
) -> String {
    let scope_sel = format!(".{}", scope);
    let pseudo_str = pseudo.as_deref().unwrap_or("");

    match on {
        Some(alias) => {
            let css_name = classes.iter()
                .find(|(a, _)| a == alias)
                .map(|(_, c)| c.as_str())
                .unwrap();
            format!("{}{} .{}", scope_sel, pseudo_str, css_name)
        }
        None => {
            format!("{}{}", scope_sel, pseudo_str)
        }
    }
}

fn to_pascal_case(s: &str) -> String {
    let mut result = String::new();
    let mut capitalize_next = true;
    for c in s.chars() {
        if c == '_' || c == '-' {
            capitalize_next = true;
        } else if capitalize_next {
            result.push(c.to_ascii_uppercase());
            capitalize_next = false;
        } else {
            result.push(c);
        }
    }
    result
}
```

- [ ] **Step 3: Implement the css! macro**

The `css!` macro takes a struct type and a block of CSS-like rules. It resolves selector names (SCOPE, class aliases, modifiers) to their CSS values, validates property names and values against the spec, and produces a `&'static str`.

```rust
// css-styled-derive/src/css_macro.rs
use proc_macro2::{TokenStream, Span};
use quote::quote;
use syn::{Error, Result, Ident, Token};
use syn::parse::{Parse, ParseStream};

use crate::fuzzy::closest_match;

/// Input: css!(StructName, { SELECTOR { prop: value; } ... })
pub struct CssMacroInput {
    struct_name: Ident,
    rules: Vec<CssRule>,
}

struct CssRule {
    selector_parts: Vec<SelectorPart>,
    declarations: Vec<CssDeclaration>,
}

enum SelectorPart {
    Name(Ident),           // SCOPE, INNER, ACTIVE — resolved to class name
    Dot,                   // . between compound selectors like SCOPE.ACTIVE
    Descendant,            // space between parts
}

struct CssDeclaration {
    property: String,
    property_span: Span,
    value: String,
    value_span: Span,
}

impl Parse for CssMacroInput {
    fn parse(input: ParseStream) -> Result<Self> {
        let struct_name: Ident = input.parse()?;
        let _comma: Token![,] = input.parse()?;

        let content;
        syn::braced!(content in input);

        let mut rules = Vec::new();
        while !content.is_empty() {
            rules.push(parse_rule(&content)?);
        }

        Ok(CssMacroInput { struct_name, rules })
    }
}

fn parse_rule(input: ParseStream) -> Result<CssRule> {
    // Parse selector parts until we hit a {
    let mut selector_parts = Vec::new();

    while !input.peek(syn::token::Brace) {
        if input.peek(Token![.]) {
            let _: Token![.] = input.parse()?;
            selector_parts.push(SelectorPart::Dot);
        } else {
            let ident: Ident = input.parse()?;
            // Check if previous was a Name (not Dot) — means descendant combinator
            if !selector_parts.is_empty() {
                if let Some(SelectorPart::Name(_)) = selector_parts.last() {
                    selector_parts.push(SelectorPart::Descendant);
                }
            }
            selector_parts.push(SelectorPart::Name(ident));
        }
    }

    // Parse declarations block
    let decl_content;
    syn::braced!(decl_content in input);

    let mut declarations = Vec::new();
    while !decl_content.is_empty() {
        declarations.push(parse_declaration(&decl_content)?);
    }

    Ok(CssRule { selector_parts, declarations })
}

fn parse_declaration(input: ParseStream) -> Result<CssDeclaration> {
    // Property name: might be hyphenated like "align-items"
    let mut property = String::new();
    let property_span = input.span();

    let first: Ident = input.parse()?;
    property.push_str(&first.to_string());

    // Consume hyphenated parts: align-items, font-weight, etc.
    while input.peek(Token![-]) {
        let _: Token![-] = input.parse()?;
        property.push('-');
        let part: Ident = input.parse()?;
        property.push_str(&part.to_string());
    }

    let _colon: Token![:] = input.parse()?;

    // Value: everything until semicolon
    let mut value = String::new();
    let value_span = input.span();

    while !input.peek(Token![;]) {
        let tt: proc_macro2::TokenTree = input.parse()?;
        if !value.is_empty() {
            value.push(' ');
        }
        value.push_str(&tt.to_string());
    }
    let _semi: Token![;] = input.parse()?;

    Ok(CssDeclaration {
        property,
        property_span,
        value,
        value_span,
    })
}

pub fn expand(input: CssMacroInput) -> Result<TokenStream> {
    // We need to know the struct's scope, classes, and modifiers.
    // Since we can't read the struct's attributes from a proc_macro (only proc_macro_derive
    // receives the struct), we need to resolve names at macro expansion time.
    //
    // The css! macro resolves names by convention:
    // - The struct must have been derived with StyledComponent, which generates
    //   associated constants (SCOPE, class aliases as uppercase, modifiers as uppercase).
    // - We validate property names and values against the spec here.
    // - Selector name resolution happens at compile time via the generated constants.

    let struct_name = &input.struct_name;
    let mut rule_strings: Vec<String> = Vec::new();
    let mut validation_errors: Vec<Error> = Vec::new();

    for rule in &input.rules {
        // Build selector string — we generate code that references the struct's constants
        // But since we need a &'static str result, we need to build the string at compile time.
        // We can't reference constants from a proc_macro... so instead we validate names
        // against the spec data and build the CSS string as a literal.
        //
        // The name resolution approach: the macro accepts UPPERCASE identifiers and
        // maps them to CSS class names. But we don't know the mapping at macro time.
        //
        // Alternative: generate code that builds the string at runtime, then leak it to
        // get &'static str. But that's wasteful.
        //
        // Best approach: the css! macro generates code that calls a const fn or uses
        // concat! on the struct's constants. But concat! doesn't work with const items.
        //
        // Pragmatic approach: generate the selector-building code as a lazy_static or
        // once_cell that runs once.

        // Validate declarations
        for decl in &rule.declarations {
            if css_spec_data::property(&decl.property).is_none() {
                let all_names = css_spec_data::all_property_names();
                let suggestion = closest_match(&decl.property, all_names);
                let msg = if let Some(s) = suggestion {
                    format!("unknown CSS property \"{}\"\n\n  = help: did you mean \"{}\"?", decl.property, s)
                } else {
                    format!("unknown CSS property \"{}\"", decl.property)
                };
                validation_errors.push(Error::new(decl.property_span, msg));
            }

            // Value validation at compile time
            let result = css_spec_data::validate_value(&decl.property, &decl.value);
            match result {
                css_spec_data::ValidationResult::Invalid(msg) => {
                    validation_errors.push(Error::new(decl.value_span, msg));
                }
                css_spec_data::ValidationResult::Warn(msg) => {
                    validation_errors.push(Error::new(decl.value_span,
                        format!("warning: {}", msg)));
                }
                _ => {}
            }
        }
    }

    if !validation_errors.is_empty() {
        let mut combined = validation_errors.remove(0);
        for e in validation_errors {
            combined.combine(e);
        }
        return Err(combined);
    }

    // Generate code that builds the CSS string using the struct's constants.
    // We generate a block that constructs the string from the struct's associated consts.
    let rule_exprs: Vec<TokenStream> = input.rules.iter().map(|rule| {
        let selector_expr = build_selector_expr(struct_name, &rule.selector_parts);
        let decl_str: String = rule.declarations.iter()
            .map(|d| format!("{}: {};", d.property, d.value))
            .collect::<Vec<_>>()
            .join(" ");

        quote! {
            parts.push(format!("{} {{ {} }}", #selector_expr, #decl_str));
        }
    }).collect();

    Ok(quote! {
        {
            static CSS: std::sync::OnceLock<String> = std::sync::OnceLock::new();
            CSS.get_or_init(|| {
                let mut parts: Vec<String> = Vec::new();
                #(#rule_exprs)*
                parts.join("\n")
            }).as_str()
        }
    })
}

fn build_selector_expr(struct_name: &Ident, parts: &[SelectorPart]) -> TokenStream {
    // Build an expression that concatenates the selector parts using the struct's constants.
    let mut pieces: Vec<TokenStream> = Vec::new();

    for part in parts {
        match part {
            SelectorPart::Name(ident) => {
                // Reference the struct's associated constant
                let const_name = ident.clone();
                pieces.push(quote! { format!(".{}", #struct_name::#const_name) });
            }
            SelectorPart::Dot => {
                // Compound selector: remove the leading dot from the next part
                // We handle this by not adding a space and stripping the dot
                pieces.push(quote! { "".to_string() });
            }
            SelectorPart::Descendant => {
                pieces.push(quote! { " ".to_string() });
            }
        }
    }

    // Simplify: build the full selector as a format expression
    // Re-process: walk parts and build selector string expression
    let mut expr_parts: Vec<TokenStream> = Vec::new();
    let mut i = 0;
    let parts_vec: Vec<&SelectorPart> = parts.iter().collect();

    while i < parts_vec.len() {
        match parts_vec[i] {
            SelectorPart::Name(ident) => {
                let const_name = ident.clone();
                // Check if next is Dot (compound)
                if i + 1 < parts_vec.len() {
                    if let SelectorPart::Dot = parts_vec[i + 1] {
                        // Compound: .scope.modifier
                        // Current is first part, skip dot, next is second part
                        if i + 2 < parts_vec.len() {
                            if let SelectorPart::Name(next_ident) = parts_vec[i + 2] {
                                let next_const = next_ident.clone();
                                expr_parts.push(quote! {
                                    format!(".{}.{}", #struct_name::#const_name, #struct_name::#next_const)
                                });
                                i += 3;
                                continue;
                            }
                        }
                    }
                }
                expr_parts.push(quote! { format!(".{}", #struct_name::#const_name) });
                i += 1;
            }
            SelectorPart::Descendant => {
                // Already handled as separator between expr_parts
                i += 1;
            }
            SelectorPart::Dot => {
                i += 1;
            }
        }
    }

    if expr_parts.len() == 1 {
        expr_parts.into_iter().next().unwrap()
    } else {
        let joined = expr_parts.iter().enumerate().map(|(idx, part)| {
            if idx == 0 {
                quote! { #part }
            } else {
                quote! { format!(" {}", #part) }
            }
        });
        quote! {
            {
                let mut sel = String::new();
                #(sel.push_str(&#joined);)*
                sel
            }
        }
    }
}
```
